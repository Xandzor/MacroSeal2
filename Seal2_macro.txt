automacro Begin_Quest {
	location prontera
	eval $::config{SealQuest2} eq 0
	exclusive 1
	run-once 1
	call {
		# Rebarev Doug
		do move prt_castle 42 145
		pause 5
		## Update Quest
		do conf SealQuest2 1
	}
}
